PyScript Square Root Calculator

Result:

from js import document import math def calculate_sqrt(event): input_element = document.getElementById("number-input") result_element = document.getElementById("result") try: value = float(input_element.value) if value < 0: result_element.innerText = "Error: negative number" else: result_element.innerText = str(math.sqrt(value)) except ValueError: result_element.innerText = "Error: invalid input" btn = document.getElementById("calc-btn") btn.addEventListener("click", calculate_sqrt)
